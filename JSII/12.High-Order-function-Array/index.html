Manipulação de Arrays em JavaScript

Método map

O método map cria um novo array com os resultados da aplicação de uma função a cada elemento do array original.

const personagens = [
    { nivel: 42, nome: "Thrall", raca: "Orc", classe: "Xamã" },
    { nivel: 28, nome: "Garrosh", raca: "Orc", classe: "Guerreiro" },
    // mais personagens...
];

let niveis = personagens.map(personagem => personagem.nivel);
let nomes = personagens.map(personagem => personagem.nome);
let racas = personagens.map(personagem => personagem.raca);
let classes = personagens.map(personagem => personagem.classe);

console.log(nomes);  // ['Thrall', 'Garrosh', ...]
console.log(niveis); // [42, 28, ...]
console.log(racas);  // ['Orc', 'Orc', ...]
console.log(classes); // ['Xamã', 'Guerreiro', ...]

Método filter

O método filter cria um novo array com todos os elementos que passam em um teste fornecido por uma função.

let orcs = personagens.filter(personagem => personagem.raca === "Orc");
let niveis35 = personagens.filter(personagem => personagem.nivel === 35);
let guerreiros = personagens.filter(personagem => personagem.classe === "Guerreiro");

console.log(orcs);      // [ { nivel: 42, nome: 'Thrall', raca: 'Orc', classe: 'Xamã' }, ... ]
console.log(niveis35);  // [ { nivel: 35, nome: 'Varok', raca: 'Orc', classe: 'Guerreiro' }, ... ]
console.log(guerreiros); // [ { nivel: 28, nome: 'Garrosh', raca: 'Orc', classe: 'Guerreiro' }, ... ]

Método reduce

O método reduce aplica uma função a um acumulador e a cada valor do array para reduzi-lo a um único valor.

let nivelTotal = personagens.reduce((total, personagem) => total + personagem.nivel, 0);
console.log(nivelTotal); // Soma total dos níveis

let racas = personagens.reduce((acumulador, personagem) => {
    if (acumulador[personagem.raca]) {
        acumulador[personagem.raca].push(personagem);
    } else {
        acumulador[personagem.raca] = [personagem];
    }
    return acumulador;
}, {});
console.log(racas);

let niveis = personagens.reduce((acumulador, personagem) => {

Método sort

O método sort ordena os elementos do array e modifica o array original. Você pode fornecer uma função de comparação para definir a ordem.

personagens.sort((a, b) => a.nivel - b.nivel);
console.log(personagens);

let personagensOrdenados = personagens.slice().sort((a, b) => a.nivel - b.nivel);
console.log(personagensOrdenados);
