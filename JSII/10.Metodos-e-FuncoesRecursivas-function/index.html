Exemplo de Método em Objeto e Funções Recursivas

Objeto Pessoa

let pessoa = {
    nome: 'Jorge',
    idade: '34',
    dizerOla() {
        console.log('Ola mundo, meu nome é ' + this.nome + ' e tenho ' + this.idade + ' anos de idade');
    }
}

pessoa.dizerOla();

Função Recursiva - Fatorial

function fatorial(num) {
    console.log('Número é ' + num);
    if (num <= 1) {
        return 1;
    } else {
        console.log(num + ' * ' + (num - 1));
        return num * fatorial(num - 1);
    }
}

console.log(fatorial(3));
